#include<iostream>
using namespace std;
//hourglass pattern
void print(int n ){
    for(int i = 1; i<=n; i++){
        char ch = 64+i;
       for(int j =1; j<=i-1; j++)
            cout<<" ";
        for(int j =1; j <= 2*n-(2*i-1); j++)
            cout<<ch++;
       cout<<endl;
    }
    for(int i = 1; i<n; i++){
        char ch = 64+n-i;
       for(int j =1; j<n-i; j++)
            cout<<" ";
        for(int j =1; j <= (2*i+1); j++)
            cout<<ch++;
       cout<<endl;
    }
}

int main(){
    print(5);
    
    return 0;
}
